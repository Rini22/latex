\begin{algorithm}[H]
\textbf{Algorithm : Bubble Sort}\\
\KwData{Array of N Integer }
\KwResult{Sorted Array }
 i=0,j=0\;
 \For{$i<(n-1)$}{
  \For{$j<(n-i-1)$}{
  \If{$a[j]>a[j+1]$}{
swap(a[j],a[j+1])\;
   }
   }
 }

\end{algorithm}
